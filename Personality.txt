    <script>
        // Simple admin password (for demo purposes; in real scenarios, use secure backend authentication)
        const ADMIN_PASSWORD = "zanityadmin123"; // Change this to a secure password

        // Load saved links from localStorage
        function loadLinks() {
            const links = JSON.parse(localStorage.getItem('creatorLinks')) || [
                { name: 'Creator 1 - Gaming Streams', url: 'https://example.com/creator1' },
                { name: 'Creator 2 - Art Tutorials', url: 'https://example.com/creator2' },
                { name: 'Creator 3 - Tech Reviews', url: 'https://example.com/creator3' }
            ];
            const ul = document.getElementById('creator-links');
            ul.innerHTML = '';
            links.forEach(link => {
                const li = document.createElement('li');
                li.innerHTML = `<a href="${link.url}" target="_blank">${link.name}</a> <button class="remove-link" style="display:none;">Remove</button>`;
                ul.appendChild(li);
            });
        }

        // Save links to localStorage
        function saveLinks() {
            const links = [];
            document.querySelectorAll('#creator-links li').forEach(li => {
                const a = li.querySelector('a');
                links.push({ name: a.textContent, url: a.href });
            });
            localStorage.setItem('creatorLinks', JSON.stringify(links));
        }

        // Admin access
        document.getElementById('admin-button').addEventListener('click', () => {
            const password = prompt('Enter admin password:');
            if (password === ADMIN_PASSWORD) {
                document.getElementById('admin-panel').style.display = 'block';
                document.querySelectorAll('.remove-link').forEach(btn => btn.style.display = 'inline');
                document.getElementById('admin-button').style.display = 'none';
            } else {
                alert('Incorrect password!');
            }
        });

        // Add link
        document.getElementById('add-link').addEventListener('click', () => {
            const name = document.getElementById('new-link-name').value;
            const url = document.getElementById('new-link-url').value;
            if (name && url) {
                const ul = document.getElementById('creator-links');
                const li = document.createElement('li');
                li.innerHTML = `<a href="${url}" target="_blank">${name}</a> <button class="remove-link">Remove</button>`;
                ul.appendChild(li);
                saveLinks();
                document.getElementById('new-link-name').value = '';
                document.getElementById('new-link-url').value = '';
            } else {
                alert('Please enter both name and URL.');
            }
        });

        // Remove link (event delegation)
        document.getElementById('creator-links').addEventListener('click', (e) => {
            if (e.target.classList.contains('remove-link')) {
                e.target.parentElement.remove();
                saveLinks();
            }
        });

        // Logout
        document.getElementById('logout-admin').addEventListener('click', () => {
            document.getElementById('admin-panel').style.display = 'none';
            document.querySelectorAll('.remove-link').forEach(btn => btn.style.display = 'none');
            document.getElementById('admin-button').style.display = 'block';
        });

        // Initial load
        loadLinks();
    </script>
