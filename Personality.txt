                ul.appendChild(li);
            });
        }

        // Events
        function loadEvents() {
            const events = JSON.parse(localStorage.getItem('events')) || [
                { name: 'COMMUNITY GAME NIGHT', date: '2026-01-10', time: '20:00', desc: 'MULTIPLAYER CHAOS' },
                { name: 'CREATOR COLLAB', date: '2026-01-20', time: '19:00', desc: 'LIVE WITH FEATURED MEMBERS' }
            ];
            const ul = document.getElementById('event-list');
            ul.innerHTML = '';
            events.forEach(event => {
                const li = document.createElement('li');
                li.innerHTML = `<strong>\( {event.name}</strong><br> \){event.date} @ \( {event.time} \){event.desc ? ' — ' + event.desc : ''}<button class="remove-link" style="display:none;">CANCEL</button>`;
                ul.appendChild(li);
            });
        }

        // P1 Members
        function loadP1s() {
            const p1s = JSON.parse(localStorage.getItem('p1s')) || [
                { name: 'NEONKNIGHT', desc: 'LEGENDARY SUPPORT & MODERATION' },
                { name: 'GRIDRUNNER', desc: 'ALWAYS ACTIVE & HELPFUL' }
            ];
            const ul = document.getElementById('p1-list');
            ul.innerHTML = '';
            p1s.forEach(p1 => {
                const li = document.createElement('li');
                li.innerHTML = `<strong>\( {p1.name}</strong><br> \){p1.desc}<button class="remove-link" style="display:none;">REVOKE</button>`;
                ul.appendChild(li);
            });
        }

        // Admin Access - Password Prompt
        document.getElementById('admin-button').addEventListener('click', () => {
            const password = prompt('ENTER ADMIN CREDENTIALS TO ACCESS TERMINAL:');
            if (password === ADMIN_PASSWORD) {
                document.getElementById('admin-panel').style.display = 'grid';
                document.querySelectorAll('.remove-link').forEach(btn => btn.style.display = 'inline-block');
                document.getElementById('admin-button').style.display = 'none';
                alert('ACCESS GRANTED — TERMINAL ACTIVE');
            } else if (password !== null) {
                alert('ACCESS DENIED — INCORRECT CREDENTIALS');
            }
        });

        // Add Creator
        document.getElementById('add-creator').addEventListener('click', () => {
            const name = document.getElementById('new-creator-name').value.trim().toUpperCase();
            const desc = document.getElementById('new-creator-desc').value.trim().toUpperCase();
            const youtube = document.getElementById('new-creator-youtube').value.trim();
            const instagram = document.getElementById('new-creator-instagram').value.trim();
            const twitch = document.getElementById('new-creator-twitch').value.trim();
            const other = document.getElementById('new-creator-other').value.trim();
            if (name && desc) {
                const creators = JSON.parse(localStorage.getItem('creators')) || [];
                creators.push({ name, desc, youtube, instagram, twitch, other });
                localStorage.setItem('creators', JSON.stringify(creators));
                loadCreators();
                document.querySelectorAll('#admin-panel input[type="text"]').forEach(i => i.value = '');
                alert('CREATOR DEPLOYED SUCCESSFULLY');
            } else {
                alert('ERROR: NAME AND DESCRIPTION REQUIRED');
            }
        });

        // Add Event
        document.getElementById('add-event').addEventListener('click', () => {
            const name = document.getElementById('new-event-name').value.trim().toUpperCase();
            const date = document.getElementById('new-event-date').value;
            const time = document.getElementById('new-event-time').value;
            const desc = document.getElementById('new-event-desc').value.trim().toUpperCase();
            if (name && date && time) {
                const events = JSON.parse(localStorage.getItem('events')) || [];
                events.push({ name, date, time, desc });
                localStorage.setItem('events', JSON.stringify(events));
                loadEvents();
                document.getElementById('new-event-name').value = '';
                document.getElementById('new-event-date').value = '';
                document.getElementById('new-event-time').value = '';
                document.getElementById('new-event-desc').value = '';
                alert('EVENT SCHEDULED');
            } else {
                alert('ERROR: TITLE, DATE, AND TIME REQUIRED');
            }
        });

        // Add P1
        document.getElementById('add-p1').addEventListener('click', () => {
            const name = document.getElementById('new-p1-name').value.trim().toUpperCase();
            const desc = document.getElementById('new-p1-desc').value.trim().toUpperCase();
            if (name && desc) {
                const p1s = JSON.parse(localStorage.getItem('p1s')) || [];
                p1s.push({ name, desc });
                localStorage.setItem('p1s', JSON.stringify(p1s));
                loadP1s();
                document.getElementById('new-p1-name').value = '';
                document.getElementById('new-p1-desc').value = '';
                alert('P1 HONORED');
            } else {
                alert('ERROR: USERNAME AND REASON REQUIRED');
            }
        });

        // Remove Items
        document.addEventListener('click', (e) => {
            if (e.target.classList.contains('remove-link')) {
                if (confirm('CONFIRM DELETION? THIS ACTION IS IRREVERSIBLE.')) {
                    e.target.parentElement.remove();
                    loadCreators();
                    loadEvents();
                    loadP1s();
                }
            }
        });

        // Logout Admin
        document.getElementById('logout-admin').addEventListener('click', () => {
            document.getElementById('admin-panel').style.display = 'none';
            document.querySelectorAll('.remove-link').forEach(btn => btn.style.display = 'none');
            document.getElementById('admin-button').style.display = 'block';
            alert('SESSION TERMINATED — TERMINAL SECURED');
        });

        // Initial Load
        loadCreators();
        loadEvents();
        loadP1s();
    </script>
